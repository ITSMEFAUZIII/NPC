━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Branching dialogue system. Drives all conversations and narration.

DIALOGUE NODE STRUCTURE:
{
  id: string,
  trigger: {
    day: number | [number, number],   // exact day or range
    hour: number | null,               // null = any hour
    location: string | null,
    flags: string[],                   // required flags to be true
    character: string | null           // who initiates
  },
  lines: [
    {
      speaker: 'NARRATOR' | 'KAEL' | 'ALDRIC' | npc_id,
      text: string,            // supports {day}, {coins} etc
      emotion: 'neutral' | 'happy' | 'sad' | 'angry' | 'bitter'
    }
  ],
  choices: [
    {
      id: string,
      text: string,            // shown to player
      condition: string | null, // flag/stat condition
      effects: {
        flags: {},              // flags to set
        stats: {},              // kael stats to change
        worldStats: {},         // world stats to change
        addItem: string | null,
        removeItem: string | null,
        startDialogue: string | null  // chain to another dialogue
      },
      nextNode: string | null   // null = end dialogue
    }
  ]
}

DIALOGUE ENGINE METHODS:
  startDialogue(id, gameState)
    - Find dialogue by id
    - Check trigger conditions
    - Emit 'dialogue:start'
    - Push to React state via dispatch
    - Lock player movement

  advance()
    - Move to next line in current node
    - If no more lines: show choices
    - If no choices: end dialogue

  selectChoice(choiceId)
    - Apply effects to gameState
    - If nextNode: load next dialogue node
    - Else: end dialogue
    - Emit 'kael:action' with choice data

  endDialogue()
    - Emit 'dialogue:end'
    - Unlock player movement
    - Clear from React state

TEXT PROCESSING:
  - Replace {day} with current day
  - Replace {coins} with kael.stats.coins
  - Replace {heroName} with hero.name
  - Replace {influence} with vague descriptor:
      0-20: "nothing"
      21-40: "a whisper"
      41-60: "a current"
      61-80: "a tide"
      81-100: "a force"
